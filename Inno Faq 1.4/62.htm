function NextButtonClick(CurPageID: Integer): Boolean;
Begin
Result:= True

Result:= MessageBox(StrToInt(ExpandConstant('{wizardhwnd}')), ExpandConstant('{cm:SysDirSelect}'), 'Установка в системную папку', MB_YESNO or $30) = idYes;
end;
