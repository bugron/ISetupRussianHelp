[Setup]
AppName=DateTimeStop
AppVerName=DateTimeStop
Uninstallable=false
UpdateUninstallLogAppName=false
DisableDirPage=true
DisableProgramGroupPage=true
DefaultDirName={pf}\DateTimeStop
DisableStartupPrompt=true

[Code]
function GetAcrobatSearchVersion(): String;
var
  sVersion:  String;
begin
  sVersion := '';
  if (RegQueryStringValue(HKLM, 'SOFTWARE\Microsoft\Windows\CurrentVersion\App Paths\AcroRd32.exe', '', sVersion)) then
  begin
    sVersion := ExtractFileDir(sVersion) + '\plug_ins\search.api';
  end;
  GetVersionNumbersString(sVersion , sVersion);
  Result := sVersion;
  MsgBox(sVersion, mbInformation, MB_OK);
end;

function InitializeSetup(): Boolean;
begin
  GetAcrobatSearchVersion();
  result:=true;
end;
