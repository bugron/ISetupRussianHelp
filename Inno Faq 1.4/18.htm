[Code]
var
CurNumber: Integer;
QuestionString, AnswerString: array of String;
ProtectionPage: TWizardPage;
ProtectionLabel, QuestionNameLabel, EditNameLabel: TLabel;
ProtectionEdit: TEdit;

function NextButtonClick(CurPageId: Integer): Boolean;
begin
Result:= True;
if CurPageId = ProtectionPage.Id then
  begin
  if ProtectionEdit.Text <> AnswerString[CurNumber] then
    begin
      MsgBox('Ответ неверен!', mb Information, MB_OK)
      Result:= False;
    end else
    Result:= True;
  end;
end;

function CM(Message: String): String;
Begin
Result:= ExpandConstant('{cm:'+ Message +'}')
end;

procedure InitializeWizard();
begin
CurNumber:= Random(15);

QuestionString:= [CM('Question0'),CM('Question1'),CM('Question2'),CM('Question3'),CM('Question4'),
CM('Question5'),CM('Question6'),CM('Question7'),CM('Question8'),CM('Question9'),CM('Question10'),
CM('Question11'), CM('Question12'),CM('Question13'),CM('Question14')];

AnswerString:= [CM('Answer0'), CM('Answer1'), CM('Answer2'),CM('Answer3'),CM('Answer4'),
CM('Answer5'),CM('Answer6'), CM('Answer7'), CM('Answer8'),CM('Answer9'),CM('Answer10'),
CM('Answer11'),CM('Answer12'), CM('Answer13'), CM('Answer14')];

ProtectionPage:= CreateCustomPage(wpWelcome, 'Лингвистическая защита', '')

EditNameLabel:= TLabel.Create(ProtectionPage)
EditNameLabel.SetBounds(ScaleX(0), ScaleY(0), ScaleX(400), ScaleY(60));
EditNameLabel.AutoSize:= False;
EditNameLabel.WordWrap:= True;
EditNameLabel.Transparent:= True;
EditNameLabel.Caption:= 'Угадайте загадку:'
EditNameLabel.Parent:= ProtectionPage.Surface

ProtectionLabel:= TLabel.Create(ProtectionPage)
ProtectionLabel.SetBounds(ScaleX(30), ScaleY(20), ScaleX(387), ScaleY(60));
ProtectionLabel.AutoSize:= False;
ProtectionLabel.WordWrap:= True;
ProtectionLabel.Transparent:= True;
ProtectionLabel.Caption:= QuestionString[CurNumber]
ProtectionLabel.Parent:= ProtectionPage.Surface

EditNameLabel:= TLabel.Create(ProtectionPage)
EditNameLabel.SetBounds(ScaleX(0), ScaleY(110), ScaleX(400), ScaleY(60));
EditNameLabel.AutoSize:= False;
EditNameLabel.WordWrap:= True;
EditNameLabel.Transparent:= True;
EditNameLabel.Caption:= 'Введите ответ строчными буквами, сделав превую заглавной.'
EditNameLabel.Parent:= ProtectionPage.Surface

ProtectionEdit:= TEdit.Create(ProtectionPage)
ProtectionEdit.SetBounds(ScaleX(0), ScaleY(130), ScaleX(417), ScaleY(60));
