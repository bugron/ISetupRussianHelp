;;Debug: no files, no compression, just test script, Send output to CATI-PP.iss
;;DebugFiles: no compression but use files, to test setup, Send output to CATI-PP.iss
;;Release: files with compression, to release smallest setup
#define Debug
;;#define DebugFiles
;;#define Release

;;flags for Release/debug
#ifdef Debug
  #define SavePPOutput
#endif

#ifdef DebugFiles
  #define SavePPOutput
  #define Files
#endif

#ifdef Release
  #define Compression
  #define Files
#endif

[Setup]
AppName=MyProgram-differentBuilds
AppVerName=MyProgram-differentBuilds v1
DefaultDirName={pf}\MyProg
#ifdef Compression
Compression=lzma2/ultra64
InternalCompressLevel=max
SolidCompression=true
#else
Compression=none
InternalCompressLevel=none
SolidCompression=false
#endif

[Files]
#ifdef Files
Source: compiler:Examples\MyProg.exe; DestDir: {app}
#endif

#ifdef SavePPOutput
  #expr SaveToFile(AddBackslash(SourcePath) + "differentBuild-PP.iss")
#endif
