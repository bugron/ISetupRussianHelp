procedure TestButtonOnClick(Sender: TObject);
begin

end;

procedure InitializeWizard();
var
NewButton1: TNewButton;

begin

with NewButton1 do begin
Left:=40
Top:=330
Width:=81
Height:=22
Cursor:=crHand
OnClick:=@TestButtonOnClick
Parent:=WizardForm
Caption := 'Инфо';
end
end;
