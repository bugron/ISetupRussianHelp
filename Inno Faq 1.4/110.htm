» Inno Setup Faq. Как обнаружить по имени класса окна выполняется ли Excel


[Setup]
AppName=ExcelRunning
AppVerName=ExcelRunning
DefaultDirName={pf}\ExcelRunning
DisableStartupPrompt=true
Uninstallable=false
DisableDirPage=true
OutputBaseFilename=ExcelRunning
CreateAppDir=false
DisableProgramGroupPage=true

[Code]
function IsExcelRunning(): Boolean;
var hWnd: Longint;
begin
 Result := false;
 hWnd := FindWindowByClassName('XLMAIN');
 if hWnd>0 then Result := true;
end;

function InitializeSetup(): Boolean;
begin
  if IsExcelRunning() then
  begin
   MsgBox('MS Excel is running' + #13 +
    'Please shutdown it and start again this setup.',
     mbError, MB_OK);
   ///cannot go with setup anyway
   Result := false;
  end
  else begin
///go on with setup
Result := true;
  end;
end;
