» Inno Setup Faq. Как обнаружить vbscript во время выполнения wscript.exe


[Setup]
AppName=WScriptVersion
AppVerName=WScriptVersion
DefaultDirName=WScriptVersion
DisableStartupPrompt=true
Uninstallable=false

[Code]
function GetWScriptVersion(): String;
var sVersion, fname:  String;
begin
  sVersion := '';
  if UsingWinNT
    then fname := ExpandConstant('{sys}\wscript.exe')
    else fname := ExpandConstant('{win}\wscript.exe');
  GetVersionNumbersString(fname , sVersion);
  Result := sVersion;
end;

function InitializeSetup(): Boolean;
begin
  MsgBox('Version of WScript.exe is (' + GetWScriptVersion() + ')',
   mbInformation, MB_OK);
  Result := false;
end;
