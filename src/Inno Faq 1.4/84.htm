function IsAnsi(S: string): boolean;
var
S1: string;
S2: string;
begin
S1 := AnsiUppercase(S);
S2 := Uppercase(S);
if CompareStr(S1, S2) = 0 then begin
S1 := Lowercase(S);
S2 := AnsiLowercase(S);
if CompareStr(S1, S2) = 0 then
Result := True;
end;
end;
